97void foc_calculate_dutycycle(float elect_angle, float d, float q, float *u, float *v, float *w) {
98 float alpha, beta;
99
100 /* fast calculation of cosine and sine value of electric angle */
101 float cf = fast_cos(elect_angle);
102 float sf = fast_sin(elect_angle);
103
104 /* firstly, the inverse Park transform is calculated */
105 alpha = d * cf - q * sf;
106 beta = q * cf + d * sf;
107
108 /* secondly, the inverse Clarke transform is calculated */
109 *u = 0.5f - alpha / VBUS;
110 *v = 0.5f + (alpha * 0.5f - beta * 0.866025404f) / VBUS;
111 *w = 1.5f - *u - *v;
